% test_chemistry.tex - Chemical equations and notation
\documentclass{article}
\usepackage{amsmath}
\usepackage{amssymb}
\pagestyle{empty}
\begin{document}

% Simple chemical equations
$$\text{H}_2 + \text{O}_2 \to \text{H}_2\text{O}$$

$$2\text{H}_2 + \text{O}_2 \to 2\text{H}_2\text{O}$$

% Balanced equation
$$\text{CH}_4 + 2\text{O}_2 \to \text{CO}_2 + 2\text{H}_2\text{O}$$

% Equilibrium
$$\text{N}_2 + 3\text{H}_2 \rightleftharpoons 2\text{NH}_3$$

% Ionic equation
$$\text{Ag}^+ + \text{Cl}^- \to \text{AgCl} \downarrow$$

% Oxidation-reduction
$$\text{Fe}^{2+} \to \text{Fe}^{3+} + e^-$$

% Equilibrium constant
$$K_\text{eq} = \frac{[\text{C}]^c [\text{D}]^d}{[\text{A}]^a [\text{B}]^b}$$

% pH
$$\text{pH} = -\log_{10}[\text{H}^+]$$

% Henderson-Hasselbalch
$$\text{pH} = \text{p}K_\text{a} + \log\frac{[\text{A}^-]}{[\text{HA}]}$$

% Nernst equation
$$E = E^\circ - \frac{RT}{nF}\ln Q$$

% Arrhenius equation
$$k = A e^{-E_\text{a}/RT}$$

% Ideal gas law
$$PV = nRT$$

% Van der Waals equation
$$\left(P + \frac{an^2}{V^2}\right)(V - nb) = nRT$$

% Gibbs free energy
$$\Delta G = \Delta H - T\Delta S$$

$$\Delta G^\circ = -RT \ln K$$

% Electron configuration notation
$$\text{Fe}: [\text{Ar}] 3d^6 4s^2$$

% Orbital notation
$$1s^2 2s^2 2p^6 3s^2 3p^6 3d^{10} 4s^2 4p^6$$

% Rate law
$$\text{Rate} = k[\text{A}]^m[\text{B}]^n$$

% Half-life
$$t_{1/2} = \frac{\ln 2}{k} = \frac{0.693}{k}$$
