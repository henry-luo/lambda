% test_nested_structures.tex - Deeply nested TeX structures
\documentclass{article}
\usepackage{amsmath}
\usepackage{amssymb}
\pagestyle{empty}
\begin{document}

% Nested fractions
$$\cfrac{1}{1 + \cfrac{1}{2 + \cfrac{1}{3 + \cfrac{1}{4 + \cfrac{1}{5}}}}}$$

% Deeply nested radicals
$$\sqrt{1 + \sqrt{2 + \sqrt{3 + \sqrt{4 + \sqrt{5}}}}}$$

% Mixed nested structures
$$\frac{\sqrt{a + \frac{b}{c + \frac{d}{e}}}}{\sqrt{f + \frac{g}{h + \frac{i}{j}}}}$$

% Nested delimiters
$$\left[\left(\left\{\left|\frac{a}{b}\right|\right\}\right)\right]$$

% Triple nested subscripts/superscripts
$$x_{i_j}^{k^{l^m}}$$

$$a_{b_{c_{d_e}}}^{f^{g^{h^i}}}$$

% Matrix inside fraction
$$\frac{\begin{pmatrix} a & b \\ c & d \end{pmatrix}}{\begin{pmatrix} e & f \\ g & h \end{pmatrix}}$$

% Sum inside product inside integral
$$\int_0^1 \prod_{i=1}^n \left( \sum_{j=1}^m a_{ij} x^j \right) dx$$

% Nested limits
$$\lim_{x \to \lim_{y \to 0} f(y)} g(x)$$

% Complex nested expression
$$\left\| \frac{\partial}{\partial x} \left[ \sum_{n=0}^{\infty} \frac{1}{n!} \left( \int_a^b f(t) \, dt \right)^n \right] \right\|$$

% Nested arrays
$$\begin{pmatrix}
\begin{array}{cc} a & b \\ c & d \end{array} &
\begin{array}{cc} e & f \\ g & h \end{array} \\
\begin{array}{cc} i & j \\ k & l \end{array} &
\begin{array}{cc} m & n \\ o & p \end{array}
\end{pmatrix}$$

% Nested cases
$$f(x) = \begin{cases}
\begin{cases} a & x > 0 \\ b & x = 0 \end{cases} & \text{if } y > 0 \\
\begin{cases} c & x > 0 \\ d & x = 0 \end{cases} & \text{if } y \leq 0
\end{cases}$$
