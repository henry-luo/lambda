% accents.tex - Test math accents
\documentclass{article}
\usepackage{amsmath}
\begin{document}

% Basic accents
\[
\hat{a} \quad \check{b} \quad \tilde{c} \quad \acute{d}
\]

\[
\grave{e} \quad \dot{f} \quad \ddot{g} \quad \breve{h}
\]

\[
\bar{i} \quad \vec{v} \quad \mathring{o}
\]

% Wide accents
\[
\widehat{xyz} \quad \widetilde{abc}
\]

% Accents on expressions
\[
\hat{x+y} \quad \bar{a+b}
\]

% Over/under constructs
\[
\overline{a+b} \quad \underline{x+y}
\]

\[
\overbrace{a+b+c}^{n} \quad \underbrace{x+y+z}_{m}
\]

% Arrows
\[
\overrightarrow{AB} \quad \overleftarrow{CD}
\]

% Stacked accents
\[
\hat{\bar{x}} \quad \dot{\hat{y}}
\]

\end{document}
