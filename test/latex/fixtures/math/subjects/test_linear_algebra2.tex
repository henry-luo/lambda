% Eigenvalue equation
$$A\vec{v} = \lambda\vec{v}$$

% Characteristic polynomial
$$\det(A - \lambda I) = 0$$

% Trace and determinant
$$\text{tr}(A) = \sum_{i=1}^{n} a_{ii}$$

$$\det(A) = \prod_{i=1}^{n} \lambda_i$$

% Inverse matrix
$$A^{-1} = \frac{1}{\det(A)} \text{adj}(A)$$
